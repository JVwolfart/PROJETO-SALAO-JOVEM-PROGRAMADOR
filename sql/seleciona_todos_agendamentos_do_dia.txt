SELECT data_agenda, hora, funcionarios.Nome AS PROFISSIONAL, clientes.Nome AS CLIENTE, clientes.Telefone, servicos.Nome AS SERVICO, clientes.Fidelizado, status_agenda  FROM agenda LEFT JOIN servicos on agenda.id_servico = servicos.Codigo LEFT JOIN clientes on agenda.id_cliente = clientes.Id_cliente LEFT JOIN funcionarios on agenda.id_profi= funcionarios.Id_func WHERE data_agenda = '2021-11-25' ORDER BY hora
