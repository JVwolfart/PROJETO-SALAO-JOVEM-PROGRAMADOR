SELECT data_agenda, hora, servicos.Tempo_medio,funcionarios.nome AS PROFISSIONAL, clientes.Nome as cliente, servicos.Nome as servico, status_agenda FROM agenda LEFT JOIN servicos on agenda.id_servico = servicos.Codigo LEFT JOIN clientes on agenda.id_cliente = clientes.Id_cliente LEFT JOIN funcionarios on agenda.id_profi= funcionarios.Id_func WHERE  data_agenda like '2021-11-%' order by data_agenda,hora
