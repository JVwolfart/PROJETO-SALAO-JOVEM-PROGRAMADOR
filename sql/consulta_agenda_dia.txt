SELECT data_agenda, hora, servicos.Tempo_medio, funcionarios.Nome as profissional, clientes.Nome as cliente, clientes.Telefone, servicos.Nome as servico, status_agenda FROM agenda LEFT JOIN servicos on agenda.id_servico = servicos.Codigo LEFT JOIN clientes on agenda.id_cliente = clientes.Id_cliente LEFT JOIN funcionarios on agenda.id_profi= funcionarios.Id_func WHERE  data_agenda >= "2021-10-26" AND status_agenda <> 'Eliminado' order by hora
